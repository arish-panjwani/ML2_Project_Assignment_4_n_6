# --- First Round of Tuning: Adjusting 'eps' ---
eps_values = [0.3, 0.5, 0.7]
best_eps = 0
best_roc_auc = 0
best_dbscan = None

# Start timing the training phase for eps tuning
start_time_eps = time.time()

# Try different eps values
for eps in eps_values:
    print(f"Tuning: eps={eps}")

    # Apply DBSCAN with current eps value
    dbscan = DBSCAN(eps=eps, min_samples=5)  # Start with a fixed min_samples of 5
    y_dbscan = dbscan.fit_predict(X_scaled)

    # Map noise (-1) to 0
    y_dbscan = np.where(y_dbscan == -1, 0, y_dbscan)

    # Calculate ROC-AUC score
    roc_auc = roc_auc_score(Y_train, y_dbscan)

    if roc_auc > best_roc_auc:
        best_roc_auc = roc_auc
        best_eps = eps
        best_dbscan = dbscan

# Stop timing the training phase for eps tuning
end_time_eps = time.time()
training_time_eps = end_time_eps - start_time_eps
print(f"Best eps: {best_eps} with ROC-AUC: {best_roc_auc}")
print(f"Time taken for eps tuning: {training_time_eps:.2f} seconds")
